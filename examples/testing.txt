Generate manifests:
go run src/github.com/IMQS/updater/updater-cmd/main.go buildmanifest c:/temp/test/server/channels/stable
go run src/github.com/IMQS/updater/updater-cmd/main.go buildmanifest c:/temp/test/server/conf/tester

go run src/github.com/IMQS/updater/server-cmd/main.go serve c:\temp\test\server

go run src/github.com/IMQS/updater/updater-cmd/main.go -config=examples\config-c-temp.json download
go run src/github.com/IMQS/updater/updater-cmd/main.go -config=examples\config-c-temp.json apply
